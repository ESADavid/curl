Integration Code

// Enhanced validation with algorithm features
validation_config_t config = {
    .base_url = "https://api.payments.jpmorgan.com/tsapi/v2",
    .timeout = 30,
    .max_retries = 3,
    .enable_caching = true,
    .enable_metrics = true,
    .enable_connection_pooling = true,
    .client_cert_path = "/path/to/cert.pem",
    .client_key_path = "/path/to/key.pem"
};

validation_set_config(&config);

validation_error_t error;
CURLcode res = curl_validation_enhanced(curl, "validations/accounts", payload, &error);

if (res == CURLE_OK) {
    printf("✅ Validation successful\n");
} else {
    printf("❌ Error: %s (Code: %d)\n", error.message, error.code);
    validation_free_error(&error);
}
